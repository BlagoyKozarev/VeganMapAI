  <script>
    let map;
    const markers = [];

    async function initMap() {
      map = new google.maps.Map(document.getElementById("map"), {
        center: { lat: 42.6977, lng: 23.3219 },
        zoom: 13,
      });

      // Симулирани ресторанти
      const locations = [
        { lat: 42.6977, lng: 23.3219, name: "Vegan Place Sofia" },
        { lat: 42.698, lng: 23.323, name: "Green Food Spot" },
        { lat: 42.699, lng: 23.319, name: "Planty Bistro" },
      ];

      for (const loc of locations) {
        const marker = new google.maps.Marker({
          position: { lat: loc.lat, lng: loc.lng },
          map,
        });

        const infoWindow = new google.maps.InfoWindow({
          content: `<strong>${loc.name}</strong><br><button onclick='alert("Още инфо скоро!")'>Детайли</button>`
        });

        marker.addListener("click", () => {
          infoWindow.open(map, marker);
        });

        markers.push(marker);
      }

      // Кластеризация
      new markerClusterer.MarkerClusterer({
        map,
        markers,
      });
    }

    window.initMap = initMap;
  </script>
