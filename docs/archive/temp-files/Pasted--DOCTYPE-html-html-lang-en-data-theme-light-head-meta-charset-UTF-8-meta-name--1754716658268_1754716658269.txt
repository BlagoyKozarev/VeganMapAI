  <script>
    const logEl = document.getElementById('log');
    const append = (msg) => {
      const ts = new Date().toLocaleTimeString();
      logEl.value = `[${ts}] ${msg}\n` + logEl.value;
    };
    document.querySelectorAll('button[data-id]').forEach(btn=>{
      btn.addEventListener('click', ()=>{
        const id = btn.dataset.id;
        const action = btn.dataset.action;
        append(`Click: ${id} → ${action}`);
        // demo behaviors (no real navigation)
        if(action === 'geo') append('• would request geolocation and center map');
        if(action === 'open-filters') append('• would open filter panel above the map');
        if(action === 'open-add') append('• would open Add place modal/form');
        if(action === 'gmaps') append('• would open Google Maps with selected place');
      });
    });
    document.getElementById('clear').addEventListener('click', ()=> logEl.value = '');

    // style presets
    document.getElementById('variant').addEventListener('change', (e)=>{
      const v = e.target.value;
      document.querySelectorAll('.btn').forEach(b=>{
        b.style.padding = (v==='compact') ? '8px 12px' : '10px 16px';
        b.style.borderWidth = (v==='outline') ? '2px' : '1px';
      });
      append(`Style preset: ${v}`);
    });
  </script>
